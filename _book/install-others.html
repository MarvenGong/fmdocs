7.5.其他

其他问题

1.修改端口

修改 ecosystem.config.js

module.exports = {
  apps: [
    {
      // 生产环境
      name: "fastmockProd",
      // 项目启动入口文件
      script: "./bin/www",
      // 项目环境变量
      env: {
        "NODE_ENV": "production",
        "PORT": 80  // 在此修改端口即可
      },
      env_production : {
        NODE_ENV: 'production',  //使用production模式 pm2 start ecosystem.config.js --env production
      },
    },
]}
